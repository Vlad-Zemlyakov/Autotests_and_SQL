SELECT с.login,
       COUNT(*) AS "inDelivery" 
FROM "Couriers" AS c
JOIN "Orders" AS о ON c.id=o."courierId" 
WHERE o."inDelivery" = 't'
GROUP BY c.login;

SELECT track,
       CASE 
           WHEN finished='t' THEN '2' 
           WHEN cancelled ='t' THEN '-1'  
           WHEN "inDelivery"='t' THEN '1' 
           ELSE '0'
       END AS status 
FROM "Orders";
